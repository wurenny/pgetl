# PGETL
This soft only for PostgreSQL

#Install
$ chmod 700 pgetl/pgetl-*

$ pgetl/pgetl-install

$ exit //means re-login


#Run demo
$ pgetl batch=1 date=sysdate

#Show etl status
$ pgetlstat -f

#Uninstall
$ pgetl/pgetl-uninstall
